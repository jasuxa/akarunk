def bubble_sort(arr):
    n = len(arr)
    
    # –ü—Ä–æ—Ö–æ–¥–∏–º–æ –ø–æ –º–∞—Å–∏–≤—É n-1 —Ä–∞–∑—ñ–≤
    for i in range(n):
        # –û—Å—Ç–∞–Ω–Ω—ñ i –µ–ª–µ–º–µ–Ω—Ç—ñ–≤ –≤–∂–µ –Ω–∞ —Å–≤–æ—ó—Ö –º—ñ—Å—Ü—è—Ö
        for j in range(0, n - i - 1):
            # –ü–æ—Ä—ñ–≤–Ω—é—î–º–æ —Å—É—Å—ñ–¥–Ω—ñ –µ–ª–µ–º–µ–Ω—Ç–∏
            if arr[j] > arr[j + 1]:
                # –ú—ñ–Ω—è—î–º–æ –º—ñ—Å—Ü—è–º–∏, —è–∫—â–æ –≤–æ–Ω–∏ –≤ –Ω–µ–ø—Ä–∞–≤–∏–ª—å–Ω–æ–º—É –ø–æ—Ä—è–¥–∫—É
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
    
    return arr
